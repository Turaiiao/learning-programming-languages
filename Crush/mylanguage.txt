def Book
    def id: int
    def title: str
    def date: str
    
    pub def () toString -> str
        return "Book {$id, $title, $date}"
    end
end

def ids: list<int> = []

def (id: int, title + date: str) create -> Book
    if ids.hasElement(id)
        return nil
    end
    return new Book{id, title, date}
end

def () main
    let bk = create(1, "The Compilers.", "2021.02.03")
    println(bk.toString())
end
