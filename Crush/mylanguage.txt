def Book
    def id: int
    def title: str
    def date: str
    
    pub def () toString -> str
        return "Book {$id, $title, $date}"
    end
end

def ids: list<int> = []

def (id: int, title + date: str) create -> Book
    cat ids.hasElement(id)
        ret nil
    elf ids.length == 0:
        ret nil
    els
        ret nil
    end
    ret new Book{id, title, date}
end

def () main
    let bk = create(1, "The Compilers.", "2021.02.03")
    std.println(bk.toString())
end
